DATAS SEGMENT
STR DB "pleaseinputastri ag:s”
STR1 DB  "0 numbers:s“
STR2 DB "1 numbers:s"
BUF DB 20
DB ?
DB 20 DUP (?)
CRLF DB OAH, ODH,"$"
X DB 30H ;存储0的个数
Y DB 30H ;存储1的个
DATAS ENDS

STACKS SEGMENT STACK
DB 200 DUP (?)
STACKS ENDS

CODES SEGMENT
ASSUME CS :CODES , DS :DATAS , SS: STACKS
START: MOV AX,DATAS
MOV DS, AX
LEA DX,STR ;显示字符串
MOV AH,9
INT 21H
MOV AH,10 ;字符电输入
LEA DX, BUF
INT 21H
M0V AH,9
INT 21H
LEA DX.CRLF ;出 行 车
MOV AH,9
INT 21H
MOV CL,BUF+1 ;第二个字节保存管守符个数
LEA SI,BUF+2;宿共仔中的地址从第3个子节开始
DEC SI
INC CL
NEXT: DEC CL
INC SI
CMP CL,0
JEVIE
MOV DL,[SI]CMP DL, 30HJE XX
CMP DL,31HJE YY
JNZ NEXT
XX:INC X
JMP NEXT
YY:INC Y
JMP NEXT

VIE: LEA DX,STR1
MOV AH,9
INT 21H
MOV BL.X
MOV DL.X
MOV AH,2
INT 21H

LEA DX,CRLF;输出换行回车
MOV AH,9
INT21H

LEA DX.STR2
MOV AH, 9
INT 21H

LEA X.5IR2
M9INT 21HHw BLY
DL.YJoy A.2

HOV AH.4CH
INT 21H

END START
